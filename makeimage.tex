\setmainfont{David CLM} 

% --- Styles ---
\tikzset{
	circleA/.style={line width=1.4pt},
	circleB/.style={line width=1pt},
	star/.style={line width=0.9pt},
	label/.style={font=\small}
}

% --- Arc text helper ---
\newcommand{\arcinscription}[4]{%
	\path[
	decorate,
	decoration={
		text along path,
		text align=center,
		raise=0.6ex,
		text={#4}
	}
	] ($(0,0)+(#2:#1)$) arc[start angle=#2, end angle=#3, radius=#1];
}

% --- Pentagram ---
\newcommand{\pentagram}[2]{%
	\path (#1) coordinate (P);
	\path
	($(P)+(90:#2)$)   coordinate (p1)
	($(P)+(162:#2)$)  coordinate (p2)
	($(P)+(234:#2)$)  coordinate (p3)
	($(P)+(306:#2)$)  coordinate (p4)
	($(P)+(18:#2)$)   coordinate (p5);
	\draw[star] (p1)--(p3)--(p5)--(p2)--(p4)--cycle;
}

% --- Hexagram ---
\newcommand{\hexagram}[2]{%
	\path (#1) coordinate (H);
	\draw[star] ($(H)+(90:#2)$)--($(H)+(210:#2)$)--($(H)+(330:#2)$)--cycle;
	\draw[star] ($(H)+(270:#2)$)--($(H)+(30:#2)$)--($(H)+(150:#2)$)--cycle;
}
