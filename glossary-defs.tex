\newglossaryentry{tetragrammaton}{
    name={Tetragrammaton},
    description={The four-letter Hebrew name of God (YHVH), considered too holy to be spoken aloud}
}

\newglossaryentry{sigil}{
    name={Sigil},
    description={A symbolic representation of a spiritual entity's desired outcome or name, used as a focal point for magical will}
}

\newglossaryentry{goetia}{
    name={Goetia},
    description={From the Greek \textit{goeteia} (sorcery/witchcraft). Refers specifically to the practice of invoking angels or demons, distinct from \textit{Theurgia} (divine work)}
}

\newglossaryentry{asalluhi}{
    name={Asallu\d{h}i},
    description={Sumerian deity of exorcism and magic, later syncretized with Marduk as the son of Enki/Ea}
}

\newglossaryentry{enuma}{
	name={En\={u}ma Eli\v{s}},
	description={An ancient Babylonian creation myth that narrates the origins of the universe and the rise of Marduk, the chief god of Babylon.}
}
