% Format: \newglossaryentry{key}{name={Shown Text}, description={Definition}}

\newglossaryentry{tetragrammaton}{
    name={Tetragrammaton},
    description={The four-letter Hebrew name of God (YHVH), considered too holy to be spoken aloud}
}

\newglossaryentry{sigil}{
    name={Sigil},
    description={A symbolic representation of a spiritual entity's desired outcome or name, used as a focal point for magical will}
}

\newglossaryentry{goetia}{
    name={Goetia},
    description={From the Greek \textit{goeteia} (sorcery/witchcraft). Refers specifically to the practice of invoking angels or demons, distinct from \textit{Theurgia} (divine work)}
}

\newglossaryentry{asalluhi}{
    name={Asallu\d{h}i},
    description={Sumerian deity of exorcism and magic, later syncretized with Marduk as the son of Enki/Ea}
}

\newglossaryentry{enuma}{
	name={En\={u}ma Eli\v{s}},
	description={An ancient Babylonian creation myth that narrates the origins of the universe and the rise of Marduk, the chief god of Babylon.}
}

\newglossaryentry{zisurru}{
	name={Zisurr\^{u}},
	description={Babylonian protective circle drawn with flour or other grain materials, used in the \textit{Maqlû} anti-witchcraft rituals. The material boundary rather than geometric precision provides the constraint}
}

\newglossaryentry{maqlu}{
	name={Maql\^{u}},
	description={Neo-Assyrian anti-witchcraft text series (lit. "Burning"). Contains ritual procedures including the \textit{zisurrû} flour circles and specifications for different grain types used against different entity classes}
}

\newglossaryentry{udughul}{
	name={Udug-\d{h}ul},
	description={Babylonian exorcism text series (lit. "Evil Demons"). Contains protective rituals and the framework for the exorcist assuming divine authority through recitation of Marduk's names}
}

\newglossaryentry{asipu}{
	name={\={A}\v{s}ipu},
	description={Babylonian exorcist-priest. Unlike petitionary prayer, the \={a}\v{s}ipu spoke \textit{as} the god during ritual, assuming divine identity to command spiritual entities}
}

\newglossaryentry{pentagrammaton}{
	name={Pentagrammaton},
	description={The five-letter divine name Y-H-Sh-V-H (Yeheshuah), inscribed at the five points of the Pentagram of Solomon. Parallel to the four-letter Tetragrammaton (YHVH)}
}

\newglossaryentry{ritualeromanum}{
	name={Rituale Romanum},
	description={The official ritual book of the Roman Catholic Church, containing the traditional rites of exorcism including prayers and formulas used against spiritual oppression}
}

\newglossaryentry{lemegeton}{
	name={Lemegeton Clavicula Salomonis},
	description={The Lesser Key of Solomon. 17th-century grimoire compiled from earlier manuscripts, containing five books including the \textit{Ars Goetia}. The standard text for Solomonic ceremonial magic}
}
